-- DATABASE MIGRATION SCRIPT FOR ARCHITECTURAL VISUALIZATION RE-ENGINE (3D BIM VIEWER)
-- Target Database: PostgreSQL 14+, Enterprise Ready, Optimized Constraints & Indexes
-- Version: 01_init.sql
-- Created At: 2026-05-20 (UTC)

-- ==========================================
-- 0. EXTENSIONS & PREREQUISITES
-- ==========================================
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- ==========================================
-- 1. CUSTOM TYPE DEFINITIONS (ENUMS)
-- ==========================================

-- Render Job Processing Status Lifecycle
CREATE TYPE render_job_status AS ENUM (
  'rendering', 
  'completed', 
  'failed', 
  'queued', 
  'paused'
);

-- Material Class Categories matching CAD/BIM standard types
CREATE TYPE material_class AS ENUM (
  'Concrete', 
  'Wood', 
  'Glass', 
  'Metal', 
  'Stone', 
  'Fabric',
  'Plaster',
  'Other'
);

-- Gallery view type classifications 
CREATE TYPE gallery_view_type AS ENUM (
  'PANORAMA', 
  'DETAIL', 
  'NIGHT', 
  'DAY', 
  'INTERIOR', 
  'EXTERIOR'
);

-- ==========================================
-- 2. SCHEMAS & DDL DEFINITIONS
-- ==========================================

-- A. Users Table (Authentication and profile links)
CREATE TABLE IF NOT EXISTS users (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  email VARCHAR(255) UNIQUE NOT NULL,
  password_hash VARCHAR(255) NOT NULL,
  first_name VARCHAR(100),
  last_name VARCHAR(100),
  role VARCHAR(50) DEFAULT 'architect',
  created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- B. Projects Table (Consolidates material assignments and rendering contexts)
CREATE TABLE IF NOT EXISTS projects (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  name VARCHAR(255) NOT NULL,
  description TEXT,
  file_url VARCHAR(1000), -- Holds reference to uploaded .ifc, .fbx, .obj models
  owner_id UUID REFERENCES users(id) ON DELETE CASCADE,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- C. PBR Materials Table (Persists custom shaders and texture coordinates)
CREATE TABLE IF NOT EXISTS materials (
  id VARCHAR(100) PRIMARY KEY, -- MAT_xxxx UUID or Code matching 3D loader indices
  title VARCHAR(255) NOT NULL,
  type material_class NOT NULL DEFAULT 'Other',
  resolution VARCHAR(100) NOT NULL DEFAULT 'PBR • 4K • Procedural',
  img VARCHAR(1000) NOT NULL, -- Texture preview map URL or CDN link
  uv_scale_u NUMERIC(6, 3) DEFAULT 1.000,
  uv_scale_v NUMERIC(6, 3) DEFAULT 1.000,
  rotation NUMERIC(6, 2) DEFAULT 0.00,
  ior NUMERIC(6, 3) DEFAULT 1.500,
  roughness NUMERIC(5, 4) DEFAULT 0.5000,
  reflectivity NUMERIC(5, 4) DEFAULT 0.50000,
  project_id UUID REFERENCES projects(id) ON DELETE SET NULL,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- D. Render Jobs Queue Table (Tracks background rendering pipeline tasks)
CREATE TABLE IF NOT EXISTS render_jobs (
  id VARCHAR(100) PRIMARY KEY, -- render-xxxx reference id
  name VARCHAR(255) NOT NULL,
  status render_job_status NOT NULL DEFAULT 'queued',
  progress INTEGER DEFAULT 0 CHECK (progress >= 0 AND progress <= 100),
  time_remaining VARCHAR(50),
  resolution VARCHAR(50) DEFAULT '4K',
  intensity NUMERIC(6, 3) DEFAULT 1.25,
  time_of_day VARCHAR(10) NOT NULL DEFAULT '12:00',
  focal_length VARCHAR(50) NOT NULL DEFAULT '35mm',
  ray_tracing BOOLEAN DEFAULT TRUE,
  img_url VARCHAR(1000) NOT NULL, -- Result image target
  date VARCHAR(50) NOT NULL, -- Snapshot calendar day
  project_id UUID REFERENCES projects(id) ON DELETE CASCADE,
  user_id UUID REFERENCES users(id) ON DELETE SET NULL,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- E. Gallery Images Table (Persisted visual outputs showcase)
CREATE TABLE IF NOT EXISTS gallery_images (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  title VARCHAR(255) NOT NULL,
  type gallery_view_type NOT NULL DEFAULT 'DAY',
  img VARCHAR(1000) NOT NULL, -- Render output CDN URL
  cols VARCHAR(100) DEFAULT 'col-span-1',
  rows VARCHAR(100) DEFAULT 'row-span-1',
  resolution VARCHAR(50) NOT NULL DEFAULT '8K',
  date VARCHAR(50) NOT NULL,
  tags TEXT[] DEFAULT '{}', -- Array of metadata tags (e.g., 'FOV: 35mm', 'Ray-Tracing: ON')
  is_rendered BOOLEAN DEFAULT FALSE,
  project_id UUID REFERENCES projects(id) ON DELETE CASCADE,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- ==========================================
-- 3. INDEXES FOR QUERY OPTIMIZATION
-- ==========================================
CREATE INDEX IF NOT EXISTS idx_users_email ON users (email);
CREATE INDEX IF NOT EXISTS idx_projects_owner ON projects (owner_id);
CREATE INDEX IF NOT EXISTS idx_materials_project ON materials (project_id);
CREATE INDEX IF NOT EXISTS idx_render_jobs_status ON render_jobs (status);
CREATE INDEX IF NOT EXISTS idx_render_jobs_project ON render_jobs (project_id);
CREATE INDEX IF NOT EXISTS idx_gallery_images_project ON gallery_images (project_id);
CREATE INDEX IF NOT EXISTS idx_gallery_images_type ON gallery_images (type);

-- ==========================================
-- 4. TRIGGER FUNCTION FOR updated_at COLUMNS
-- ==========================================
CREATE OR REPLACE FUNCTION update_timestamp()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = CURRENT_TIMESTAMP;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Apply Triggers
CREATE TRIGGER tr_users_updated_at BEFORE UPDATE ON users FOR EACH ROW EXECUTE FUNCTION update_timestamp();
CREATE TRIGGER tr_projects_updated_at BEFORE UPDATE ON projects FOR EACH ROW EXECUTE FUNCTION update_timestamp();
CREATE TRIGGER tr_materials_updated_at BEFORE UPDATE ON materials FOR EACH ROW EXECUTE FUNCTION update_timestamp();
CREATE TRIGGER tr_render_jobs_updated_at BEFORE UPDATE ON render_jobs FOR EACH ROW EXECUTE FUNCTION update_timestamp();
